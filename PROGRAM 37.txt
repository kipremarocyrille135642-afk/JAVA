/*Write a java program to add, subtract a days/month into current date
and time*\

import java.util.Calendar;
import java.text.SimpleDateFormat;

public class DateAddSubtractDemo {
    public static void main(String[] args) {
        Calendar cal = Calendar.getInstance();
        SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");

        System.out.println("Current Date & Time : " + sdf.format(cal.getTime()));

        // Ajouter des jours
        cal.add(Calendar.DAY_OF_MONTH, 10);
        System.out.println("After adding 10 days : " + sdf.format(cal.getTime()));

        // Soustraire des jours
        cal.add(Calendar.DAY_OF_MONTH, -15);
        System.out.println("After subtracting 15 days : " + sdf.format(cal.getTime()));

        // Ajouter des mois
        cal.add(Calendar.MONTH, 3);
        System.out.println("After adding 3 months : " + sdf.format(cal.getTime()));

        // Soustraire des mois
        cal.add(Calendar.MONTH, -5);
        System.out.println("After subtracting 5 months : " + sdf.format(cal.getTime()));

        // Ajouter des années
        cal.add(Calendar.YEAR, 1);
        System.out.println("After adding 1 year : " + sdf.format(cal.getTime()));
    }
}
