/*Write a java program to display date in different format*\

import java.text.SimpleDateFormat;
import java.util.Date;

public class DateFormatDemo {
    public static void main(String[] args) {
        Date today = new Date();

        SimpleDateFormat format1 = new SimpleDateFormat("dd-MM-yyyy");
        SimpleDateFormat format2 = new SimpleDateFormat("MM/dd/yyyy");
        SimpleDateFormat format3 = new SimpleDateFormat("yyyy.MM.dd");
        SimpleDateFormat format4 = new SimpleDateFormat("dd MMMM yyyy");
        SimpleDateFormat format5 = new SimpleDateFormat("EEEE, dd MMMM yyyy");
        SimpleDateFormat format6 = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");

        System.out.println("Format 1 : " + format1.format(today));
        System.out.println("Format 2 : " + format2.format(today));
        System.out.println("Format 3 : " + format3.format(today));
        System.out.println("Format 4 : " + format4.format(today));
        System.out.println("Format 5 : " + format5.format(today));
        System.out.println("Format 6 : " + format6.format(today));
    }
}
