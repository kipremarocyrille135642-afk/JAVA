/* Write a java program to use Gregorian calendar to display
calendar information*\

import java.util.GregorianCalendar;

public class GregorianCalendarDemo {
    public static void main(String[] args) {
        GregorianCalendar gc = new GregorianCalendar();

        System.out.println("Year : " + gc.get(GregorianCalendar.YEAR));
        System.out.println("Month (0-based) : " + gc.get(GregorianCalendar.MONTH));
        System.out.println("Day of Month : " + gc.get(GregorianCalendar.DAY_OF_MONTH));
        System.out.println("Day of Week : " + gc.get(GregorianCalendar.DAY_OF_WEEK));
        System.out.println("Day of Year : " + gc.get(GregorianCalendar.DAY_OF_YEAR));
        System.out.println("Hour (24-hour) : " + gc.get(GregorianCalendar.HOUR_OF_DAY));
        System.out.println("Minute : " + gc.get(GregorianCalendar.MINUTE));
        System.out.println("Second : " + gc.get(GregorianCalendar.SECOND));
        System.out.println("Is Leap Year ? : " + gc.isLeapYear(gc.get(GregorianCalendar.YEAR)));
    }
